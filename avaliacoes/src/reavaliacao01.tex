def metodo(lista,num)
  tam = lista.length

  lista.sort!

  esq = 0
  dir = tam - 1

  while esq < dir
    if (lista[esq] + lista[dir] == num)
      return true
    elsif (lista[esq] + lista[dir] > num)
      dir -= 1
    else
      esq += 1
    end
  end

  return false
end
